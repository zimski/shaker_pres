Les modules "variable" et "IF"

dcm4chee_exist=`dpkg --list | grep dcm4chee | wc -l`
$$ sk_console.get DCM
## test if dcm4chee is installed
$$ rbox.cmd_if COND

DCM:
    - dcm4chee_exist
COND:
    if: dcm4chee_exist == 0
    then: |
        $$ sk_ssh.expect CMD_dpkg
    else: |
        echo "DCM4CHEE INSTALLED"
        /etc/rc.local stop
